229     private void deleteActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_deleteActionPerformed
230         int[] indices = list.getSelectedIndices();
231         if (indices.length == 0) {
232             printMessage("No playlist selected");
233             return;
234         }
235         printMessage("Deleting...");
236         disableButtons(true);
237         cursor(true);
238         MainLogic.delete(indices);
239         cursor(false);
240         disableButtons(false);
241     }//GEN-LAST:event_deleteActionPerformed

251     private void shuffleActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_shuffleActionPerformed
252         int[] indices = list.getSelectedIndices();
253         if (indices.length == 0) {
254             printMessage("No playlist selected");
255             return;
256         }
257         printMessage("Shuffling...");
258         cursor(true);
259         disableButtons(true);
260         MainLogic.shuffle(indices, !shuffleExisting.isSelected());
261         shuffle.setEnabled(true);
262         disableButtons(false);
263         cursor(false);
264     }//GEN-LAST:event_shuffleActionPerformed

334     /**
335      * Switches between login screen and playlist screen
336      * @param b
337      */
338     public static void playListTab(boolean b) {
339         if (b) {
340             logInPanel.setEnabled(false);
341             logInPanel.setVisible(false);
342             playlistPanel.setEnabled(true);
343             playlistPanel.setVisible(true);
344         } else {
345             logInPanel.setEnabled(true);
346             logInPanel.setVisible(true);
347             playlistPanel.setEnabled(false);
348             playlistPanel.setVisible(false);
349         }
350     }

353      * Enables/disables waiting cursor

356     private void cursor(boolean b) {
357         if (b) {

359         } else {
360             setCursor(Cursor.getDefaultCursor());
361         }
