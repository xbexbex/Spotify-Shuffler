188     public static void printMessage(String m) {
189         if (returnLog().equals("")) {
190             msgArea.setText(m);
191         } else {
192             msgArea.setText(msgArea.getText() + System.getProperty("line.separator") + m);
193         }
194     }
195 
196     public static String returnLog() {
197         return msgArea.getText();
198     }
199 
200     public static String returnLastLine() {
201         String[] l = msgArea.getText().split(System.getProperty("line.separator"));
202         return l[l.length - 1];
203     }

218     public static void playListTab(boolean b) {
219         if (b) {
220             logInPanel.setEnabled(false);
221             logInPanel.setVisible(false);
222             playlistPanel.setEnabled(true);
223             playlistPanel.setVisible(true);
224         } else {
225             playlistPanel.setEnabled(false);
226             playlistPanel.setVisible(false);
227             logInPanel.setEnabled(true);
228             logInPanel.setVisible(true);
229         }
230     }
