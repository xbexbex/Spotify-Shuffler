6   package spotifysystem.logic;
7   
8   
9   import java.io.IOException;
10  import java.io.OutputStream;
11  import java.net.InetSocketAddress;
12  
13  import com.sun.net.httpserver.HttpExchange;
14  import com.sun.net.httpserver.HttpHandler;
15  import com.sun.net.httpserver.HttpServer;
16  import java.net.URI;
17  
18  public class Server {
19  
20      public static void start() throws Exception {
21          HttpServer server = HttpServer.create(new InetSocketAddress(8888), 0);
22          server.createContext("/callback", new MyHandler());
23          server.setExecutor(null);
24          server.start();
25      }
26  
27      static class MyHandler implements HttpHandler {
28          @Override
29          public void handle(HttpExchange t) throws IOException {
30              URI u = t.getRequestURI();
31              String response = u.toString();
32              t.sendResponseHeaders(200, response.length());
33              OutputStream os = t.getResponseBody();
34              os.write(response.getBytes());
35              os.close();
36          }
37      }
38  
39  }
