39      /**
40       * Gets authorization tokens from spotify using a code
41       * @param s authorization code
42       * @return error code, 0 if successfull
43       */
44      public static int getTokens(String s) {
45          status = 0;
46          code = s;
47          api = returnApi(1);
48          final SettableFuture<AuthorizationCodeCredentials> authorizationCodeCredentialsFuture = api.authorizationCodeGrant(code).build().getAsync();
49          Futures.addCallback(authorizationCodeCredentialsFuture, new FutureCallback<AuthorizationCodeCredentials>() {
50              @Override
51              public void onSuccess(AuthorizationCodeCredentials authorizationCodeCredentials) {
52                  accessToken = authorizationCodeCredentials.getAccessToken();
53                  refreshToken = authorizationCodeCredentials.getRefreshToken();
54                  status = authorizationCodeCredentials.getExpiresIn();
55                  if (timer != null) {
56                      timer.restart(status);
57                  } else {
58                      timer = new AuthTimer();
59                      timer.start(status);
60                  }
61              }
62  
63              @Override
64              public void onFailure(Throwable throwable) {
65                  status = 1;
66                  MainLogic.print(throwable.getMessage());
67              }
68  
69          });
70          return status;
71      }

180      /**
181      * Finds a suitable non-duplicate name for the new playlist by checking the list of playlists
182      * @param n String original playlist's name
183      * @return suitable name
184      */
185     private static String getSuitableName(String n) {
186         String[] nwords = n.split(" ");
187         String[] names = MainLogic.getPlaylistNames();
188         ArrayList<String> matching = new ArrayList();
189         String ln = nwords[nwords.length - 1];
190         int st = 2;
191         n = "";
192         if (ln.equals("(Shuffled)")) {
193             for (int i = 0; i < nwords.length - 1; i++) {
194                 n += nwords[i];
195                 n += " ";
196             }
197         } else if (nwords.length >= 2 && nwords[nwords.length - 2].equals("(Shuffled") && ln.charAt(ln.length() - 1) == ')') {
198             try {
199                 st = Integer.parseInt(ln.substring(0, ln.length() - 1)) + 1;
200                 for (int i = 0; i < nwords.length - 2; i++) {
201                     n += nwords[i];
202                     n += " ";
203                 }
204             } catch (NumberFormatException nfe) {
205                 for (int i = 0; i < nwords.length; i++) {
206                     n += nwords[i];
207                     n += " ";
208                 }
209             }
210         } else {
211             for (int i = 0; i < nwords.length; i++) {
212                 n += nwords[i];
213                 n += " ";
214             }
215         }
216         ArrayList<Integer> forbidden = new ArrayList();
217         for (int i = 0; i < names.length; i++) {
218             nwords = names[i].split(" ");
219             if (nwords.length >= 1 && nwords[nwords.length - 1].equals("(Shuffled)")) {
220                 forbidden.add(-1);
221             } else if (nwords.length >= 2 && nwords[nwords.length - 2].equals("(Shuffled") && nwords[nwords.length - 1].charAt(nwords[nwords.length - 1].length() - 1) == ')') {
222                 try {
223                     forbidden.add(Integer.parseInt(nwords[nwords.length - 1].substring(0, nwords[nwords.length - 1].length() - 1)));
224                 } catch (NumberFormatException nfe) {
225                 }
226             }
227         }
228         if (!forbidden.contains(-1)) {
229             return n + "(Shuffled)";
230         }
231         for (int f : forbidden) {
232             if (forbidden.contains(st)) {
233                 st++;
234             }
235         }
236         return n + "(Shuffled " + st + ")";

240      /**
241      * Randomizes the order of a list of tracks and prepares it for further use
242      * @param pltz list to be shuffled
243      * @return shuffled list
244      */
245     private static List<String> randomize(List<PlaylistTrack> pltz) {
246         PlaylistTrack[] plt = new PlaylistTrack[pltz.size()];
247         int j = 0;
248         for (PlaylistTrack t : pltz) {
249             plt[j] = t;
250             j++;
251         }
252         List<String> pltx = new ArrayList();
253         int n = plt.length;
254         for (int i = 0; i < n; i++) {
255             int r = i + (int) (Math.random() * (n - i));
256             PlaylistTrack x = plt[r];
257             plt[r] = plt[i];
258             plt[i] = x;
259         }
260         for (int i = 0; i < plt.length; i++) {
261             pltx.add(plt[i].getTrack().getUri());
262         }
263         return pltx;
264     }

273         try {

277                         .clientSecret(clientSecret)
278                         .redirectURI(redirectUri)
279                         .build();
280             } else if (type == 2) {
281                 return Api.builder()
282                         .clientId(clientId)
283                         .clientSecret(clientSecret)
284                         .build();
285             } else if (type == 3) {
286                 return Api.builder().accessToken(accessToken).build();
287             }
288         } catch (Exception e) {
289             MainLogic.print(e.getMessage());
290         }
291         return null;
