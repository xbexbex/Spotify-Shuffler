51      /**
52       * Gets authorization tokens from spotify using a code
53       *
54       * @param s authorization code
55       * @return error code, 0 if successfull
56       */
57      public static int getTokens(String s) {
58          status = 0;
59          code = s;
60          api = returnApi(1);
61          final SettableFuture<AuthorizationCodeCredentials> authorizationCodeCredentialsFuture = api.authorizationCodeGrant(code).build().getAsync();
62          Futures.addCallback(authorizationCodeCredentialsFuture, new FutureCallback<AuthorizationCodeCredentials>() {
63              @Override
64              public void onSuccess(AuthorizationCodeCredentials authorizationCodeCredentials) {
65                  accessToken = authorizationCodeCredentials.getAccessToken();
66                  refreshToken = authorizationCodeCredentials.getRefreshToken();
67                  status = authorizationCodeCredentials.getExpiresIn();
68                  if (timer != null) {
69                      timer.restart(status);
70                  } else {
71                      timer = new AuthTimer();
72                      timer.start(status);
73                  }
74              }
75  
76              @Override
77              public void onFailure(Throwable throwable) {
78                  status = 1;
79                  MainLogic.print("Something went wrong");
80              }
81  
82          });
83          return status;
84      }

207     /**
208      * Finds a suitable non-duplicate name for the new playlist by checking the
209      * list of playlists
210      *
211      * @param n String original playlist's name
212      * @return suitable name
213      */
214     private static String getSuitableName(String n) {
215         String[] nwords = n.split(" ");
216         String[] names = MainLogic.getPlaylistNames();
217         String ln = nwords[nwords.length - 1];
218         int st = 2;
219         n = "";
220         if (ln.equals("(Shuffled)")) {
221             for (int i = 0; i < nwords.length - 1; i++) {
222                 n += nwords[i];
223                 n += " ";
224             }
225         } else if (nwords.length >= 2 && nwords[nwords.length - 2].equals("(Shuffled") && ln.charAt(ln.length() - 1) == ')') {
226             try {
227                 st = Integer.parseInt(ln.substring(0, ln.length() - 1)) + 1;
228                 for (int i = 0; i < nwords.length - 2; i++) {
229                     n += nwords[i];
230                     n += " ";
231                 }
232             } catch (NumberFormatException nfe) {
233                 for (int i = 0; i < nwords.length; i++) {
234                     n += nwords[i];
235                     n += " ";
236                 }
237             }
238         } else {
239             for (int i = 0; i < nwords.length; i++) {
240                 n += nwords[i];
241                 n += " ";
242             }
243         }
244         ArrayList<Integer> forbidden = new ArrayList();
245         for (int i = 0; i < names.length; i++) {
246             nwords = names[i].split(" ");
247             if (nwords.length >= 1 && nwords[nwords.length - 1].equals("(Shuffled)")) {
248                 forbidden.add(-1);
249             } else if (nwords.length >= 2 && nwords[nwords.length - 2].equals("(Shuffled") && nwords[nwords.length - 1].charAt(nwords[nwords.length - 1].length() - 1) == ')') {
250                 try {
251                     forbidden.add(Integer.parseInt(nwords[nwords.length - 1].substring(0, nwords[nwords.length - 1].length() - 1)));
252                 } catch (NumberFormatException nfe) {
253                 }
254             }
255         }
256         if (!forbidden.contains(-1)) {
257             return n + "(Shuffled)";
258         }
259         for (int f : forbidden) {
260             if (forbidden.contains(st)) {
261                 st++;
262             }
263         }
264         return n + "(Shuffled " + st + ")";
265 
266     }

268     /**
269      * Randomizes the order of a list of tracks and prepares it for further use
270      *
271      * @param pltz list to be shuffled
272      * @return shuffled list
273      */
274     private static List<String> randomize(List<PlaylistTrack> pltz) {
275         PlaylistTrack[] plt = new PlaylistTrack[pltz.size()];
276         int j = 0;
277         for (PlaylistTrack t : pltz) {
278             plt[j] = t;
279             j++;
280         }
281         List<String> pltx = new ArrayList();
282         int n = plt.length;
283         for (int i = 0; i < n; i++) {
284             int r = i + (int) (Math.random() * (n - i));
285             PlaylistTrack x = plt[r];
286             plt[r] = plt[i];
287             plt[i] = x;
288         }
289         String url = "";
290         for (int i = 0; i < plt.length; i++) {
291             url = plt[i].getTrack().getUri();
292             if (url.contains("spotify:local:")) {
293                 localFiles++;
294             } else {
295                 pltx.add(plt[i].getTrack().getUri());
296             }
297         }
298         return pltx;
299     }

332      * @return

334     public static int removePlaylist(String plId) {
335         HttpURLConnection connection = null;
336         try {
337             URL url = new URL("https://api.spotify.com/v1/users/" + MainLogic.getUsername() + "/playlists/" + plId + "/followers");
338             connection = (HttpURLConnection) url.openConnection();
339             connection.setRequestMethod("DELETE");
340             connection.setRequestProperty("Authorization",
341                     "Bearer " + accessToken);
342             connection.setUseCaches(false);
343             connection.setDoOutput(true);
344             DataOutputStream wr = new DataOutputStream(
345                     connection.getOutputStream());
346             wr.close();
347 
348             InputStream is = connection.getInputStream();
349             BufferedReader rd = new BufferedReader(new InputStreamReader(is));
350             StringBuilder response = new StringBuilder();
351             String line;
352             while ((line = rd.readLine()) != null) {
353                 response.append(line);
354                 response.append('\r');
355             }
356             rd.close();
357             return 0;
358         } catch (Exception e) {
359             return 1;
360         } finally {
361             if (connection != null) {
362                 connection.disconnect();
363             }
364         }
365     }

392         while (true) {
393             try {
394                 if (index < 0) {
395                     break;
396                 }
397                 int c = 0;
398                 for (int i = index; i >= 0; i--) {
399                     index--;
400                     positions.add(new PlaylistTrackPosition(plt.get(index)));
401                     c++;
402                     if (c > 99) {
403                         break;
404                     }
405                 }
406                 RemoveTrackFromPlaylistRequest request = api.removeTrackFromPlaylist(MainLogic.getUsername(), id, positions).tracks(positions).build();
407                 try {
408                     Thread.sleep(3000);

410                 }
411                 request.get();
412                 positions.clear();
413             } catch (Exception t) {
414                 try {
415                     Thread.sleep(3000);
416                 } catch (Exception e) {
417                 }
